SET(KIT Actuators)
SET(UKIT ACTUATORS)

##
## Version stuff
##
SET(OPENSIM_RDACTUATORS_LIBRARY_NAME rd${KIT})
SET(OPENSIM_RDACTUATORS_MAJOR_VERSION 0)
SET(OPENSIM_RDACTUATORS_MINOR_VERSION 5)
SET(OPENSIM_RDACTUATORS_BUILD_VERSION 1)
SET(OPENSIM_RDACTUATORS_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_RDACTUATORS_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_RDACTUATORS_LIBRARY_NAME=${OPENSIM_RDACTUATORS_LIBRARY_NAME}
-DOPENSIM_RDACTUATORS_MAJOR_VERSION=${OPENSIM_RDACTUATORS_MAJOR_VERSION}
-DOPENSIM_RDACTUATORS_MINOR_VERSION=${OPENSIM_RDACTUATORS_MINOR_VERSION}
-DOPENSIM_RDACTUATORS_BUILD_VERSION=${OPENSIM_RDACTUATORS_BUILD_VERSION}
-DOPENSIM_RDACTUATORS_COPYRIGHT_YEARS="${OPENSIM_RDACTUATORS_COPYRIGHT_YEARS}"
-DOPENSIM_RDACTUATORS_AUTHORS="${OPENSIM_RDACTUATORS_AUTHORS}"
-DOPENSIM_RDACTUATORS_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} 
		    ${OpenSim_SOURCE_DIR}/Vendors 
		    ${OpenSim_SOURCE_DIR}/Vendors/xerces-c-src2_4_0/src
		    ${OpenSim_SOURCE_DIR}/Tools)

SET(EXPORT_MACRO RD${UKIT}_EXPORTS)

LINK_LIBRARIES( rdTools rdSimulation)

ADD_LIBRARY(rd${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

SET_TARGET_PROPERTIES(rd${KIT} PROPERTIES
   DEFINE_SYMBOL ${EXPORT_MACRO}
)

INSTALL_TARGETS(/lib/ RUNTIME_DIRECTORY /lib rd${KIT})
INSTALL_FILES(/include/OpenSim/${KIT} .h ${INCLUDE_FILES})
