SET(KIT Actuators)
SET(UKIT ACTUATORS)

##
## Version stuff
##
SET(OPENSIM_OSIMACTUATORS_LIBRARY_NAME osim${KIT})
SET(OPENSIM_OSIMACTUATORS_MAJOR_VERSION 2)
SET(OPENSIM_OSIMACTUATORS_MINOR_VERSION 2)
SET(OPENSIM_OSIMACTUATORS_BUILD_VERSION 0)
SET(OPENSIM_OSIMACTUATORS_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_OSIMACTUATORS_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_OSIMACTUATORS_LIBRARY_NAME=${OPENSIM_OSIMACTUATORS_LIBRARY_NAME}
-DOPENSIM_OSIMACTUATORS_MAJOR_VERSION=${OPENSIM_OSIMACTUATORS_MAJOR_VERSION}
-DOPENSIM_OSIMACTUATORS_MINOR_VERSION=${OPENSIM_OSIMACTUATORS_MINOR_VERSION}
-DOPENSIM_OSIMACTUATORS_BUILD_VERSION=${OPENSIM_OSIMACTUATORS_BUILD_VERSION}
-DOPENSIM_OSIMACTUATORS_COPYRIGHT_YEARS="${OPENSIM_OSIMACTUATORS_COPYRIGHT_YEARS}"
-DOPENSIM_OSIMACTUATORS_AUTHORS="${OPENSIM_OSIMACTUATORS_AUTHORS}"
-DOPENSIM_OSIMACTUATORS_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} 
		    ${OpenSim_SOURCE_DIR}/Vendors 
		    ${OpenSim_SOURCE_DIR}/Tools)

SET(EXPORT_MACRO OSIM${UKIT}_EXPORTS)

LINK_LIBRARIES( osimCommon osimSimulation )

ADD_LIBRARY(osim${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

SET_TARGET_PROPERTIES(osim${KIT} PROPERTIES
   DEFINE_SYMBOL ${EXPORT_MACRO}
   PROJECT_LABEL "Libraries - osim${KIT}"
)

IF(WIN32)
	INSTALL_TARGETS(/sdk/lib/ RUNTIME_DIRECTORY /bin osim${KIT})
ELSE(WIN32)
	INSTALL_TARGETS(/bin/ RUNTIME_DIRECTORY /bin osim${KIT})
ENDIF(WIN32)

INSTALL_FILES(/sdk/include/OpenSim/${KIT} .h ${INCLUDE_FILES})

IF(BUILD_TESTING)
	SUBDIRS(Test)
ENDIF(BUILD_TESTING)
