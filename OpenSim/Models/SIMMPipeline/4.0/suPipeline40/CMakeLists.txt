SET(OPENSIM_SUPIPELINE40_LIBRARY_NAME su${KIT})

ADD_DEFINITIONS(-DOPENSIM_SUPIPELINE40_LIBRARY_NAME=${OPENSIM_SUPIPELINE40_LIBRARY_NAME}
