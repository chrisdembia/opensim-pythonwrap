SET(KIT Analyses)
SET(UKIT ANALYSES)

##
## Version stuff
##
SET(OPENSIM_SUANALYSES_LIBRARY_NAME su${KIT})
SET(OPENSIM_SUANALYSES_MAJOR_VERSION 0)
SET(OPENSIM_SUANALYSES_MINOR_VERSION 7)
SET(OPENSIM_SUANALYSES_BUILD_VERSION 1)
SET(OPENSIM_SUANALYSES_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_SUANALYSES_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_SUANALYSES_LIBRARY_NAME=${OPENSIM_SUANALYSES_LIBRARY_NAME}
-DOPENSIM_SUANALYSES_MAJOR_VERSION=${OPENSIM_SUANALYSES_MAJOR_VERSION}
-DOPENSIM_SUANALYSES_MINOR_VERSION=${OPENSIM_SUANALYSES_MINOR_VERSION}
-DOPENSIM_SUANALYSES_BUILD_VERSION=${OPENSIM_SUANALYSES_BUILD_VERSION}
-DOPENSIM_SUANALYSES_COPYRIGHT_YEARS="${OPENSIM_SUANALYSES_COPYRIGHT_YEARS}"
-DOPENSIM_SUANALYSES_AUTHORS="${OPENSIM_SUANALYSES_AUTHORS}"
-DOPENSIM_SUANALYSES_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} 
		    ${OpenSim_SOURCE_DIR}/Vendors 
		    ${XERCES_INCLUDE_DIR}
		    ${OpenSim_SOURCE_DIR}/Tools)

SET(EXPORT_MACRO SU${UKIT}_EXPORTS)

LINK_LIBRARIES( debug rdTools_d optimized rdTools
				debug rdSimulation_d optimized rdSimulation
				debug rdSQP_d optimized rdSQP)

ADD_LIBRARY(su${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

SET_TARGET_PROPERTIES(su${KIT} PROPERTIES
   DEFINE_SYMBOL ${EXPORT_MACRO}
)

INSTALL_TARGETS(/lib/ RUNTIME_DIRECTORY /lib su${KIT})
INSTALL_FILES(/include/OpenSim/${KIT} .h ${INCLUDE_FILES})
