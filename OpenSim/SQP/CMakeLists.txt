SET(KIT SQP)
SET(UKIT SQP)

##
## Version stuff
##
SET(OPENSIM_RDSQP_LIBRARY_NAME rd${KIT})
SET(OPENSIM_RDSQP_MAJOR_VERSION 0)
SET(OPENSIM_RDSQP_MINOR_VERSION 6)
SET(OPENSIM_RDSQP_BUILD_VERSION 1)
SET(OPENSIM_RDSQP_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_RDSQP_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_RDSQP_LIBRARY_NAME=${OPENSIM_RDSQP_LIBRARY_NAME}
-DOPENSIM_RDSQP_MAJOR_VERSION=${OPENSIM_RDSQP_MAJOR_VERSION}
-DOPENSIM_RDSQP_MINOR_VERSION=${OPENSIM_RDSQP_MINOR_VERSION}
-DOPENSIM_RDSQP_BUILD_VERSION=${OPENSIM_RDSQP_BUILD_VERSION}
-DOPENSIM_RDSQP_COPYRIGHT_YEARS="${OPENSIM_RDSQP_COPYRIGHT_YEARS}"
-DOPENSIM_RDSQP_AUTHORS="${OPENSIM_RDSQP_AUTHORS}"
-DOPENSIM_RDSQP_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} ${OpenSim_SOURCE_DIR}/Vendors)

# __STDC__ already defined on linux
IF(WIN32)
ADD_DEFINITIONS(-DRDSQP_EXPORTS -D_USRDLL -D__STDC__)
ELSE(WIN32)
ADD_DEFINITIONS(-DRDSQP_EXPORTS -D_USRDLL)
ENDIF(WIN32)

LINK_LIBRARIES( suCFSQP rdTools)

ADD_LIBRARY(rd${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 


INSTALL_TARGETS(/lib/ RUNTIME_DIRECTORY /lib rd${KIT})
INSTALL_FILES(/include/OpenSim/${KIT} .h ${INCLUDE_FILES})
