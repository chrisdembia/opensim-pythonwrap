SET(KIT SQP)
SET(UKIT SQP)

##
## Version stuff
##
SET(OPENSIM_OSIMSQP_LIBRARY_NAME osim${KIT})
SET(OPENSIM_OSIMSQP_MAJOR_VERSION 0)
SET(OPENSIM_OSIMSQP_MINOR_VERSION 7)
SET(OPENSIM_OSIMSQP_BUILD_VERSION 1)
SET(OPENSIM_OSIMSQP_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_OSIMSQP_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_OSIMSQP_LIBRARY_NAME=${OPENSIM_OSIMSQP_LIBRARY_NAME}
-DOPENSIM_OSIMSQP_MAJOR_VERSION=${OPENSIM_OSIMSQP_MAJOR_VERSION}
-DOPENSIM_OSIMSQP_MINOR_VERSION=${OPENSIM_OSIMSQP_MINOR_VERSION}
-DOPENSIM_OSIMSQP_BUILD_VERSION=${OPENSIM_OSIMSQP_BUILD_VERSION}
-DOPENSIM_OSIMSQP_COPYRIGHT_YEARS="${OPENSIM_OSIMSQP_COPYRIGHT_YEARS}"
-DOPENSIM_OSIMSQP_AUTHORS="${OPENSIM_OSIMSQP_AUTHORS}"
-DOPENSIM_OSIMSQP_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} ${OpenSim_SOURCE_DIR}/Vendors)

# __STDC__ already defined on linux
IF(WIN32)
ADD_DEFINITIONS(-DOSIMSQP_EXPORTS -D__STDC__)
ELSE(WIN32)
ADD_DEFINITIONS(-DOSIMSQP_EXPORTS)
ENDIF(WIN32)

LINK_LIBRARIES( osimCFSQP osimCommon ${SIMTK_COMMON_LIB} ${SIMTK_MATH_LIB})

ADD_LIBRARY(osim${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 


INSTALL_TARGETS(/lib/ RUNTIME_DIRECTORY /bin osim${KIT})
INSTALL_FILES(/include/OpenSim/${KIT} .h ${INCLUDE_FILES})
