SET(KIT SimmKinematicsEngine)
SET(UKIT SIMMKINEMATICSENGINE)

##
## Version stuff
##
SET(OPENSIM_SIMMKINEMATICSENGINE_LIBRARY_NAME osim${KIT})
SET(OPENSIM_SIMMKINEMATICSENGINE_MAJOR_VERSION 0)
SET(OPENSIM_SIMMKINEMATICSENGINE_MINOR_VERSION 7)
SET(OPENSIM_SIMMKINEMATICSENGINE_BUILD_VERSION 1)
SET(OPENSIM_SIMMKINEMATICSENGINE_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_SIMMKINEMATICSENGINE_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_SIMMKINEMATICSENGINE_LIBRARY_NAME=${OPENSIM_SIMMKINEMATICSENGINE_LIBRARY_NAME}
-DOPENSIM_SIMMKINEMATICSENGINE_MAJOR_VERSION=${OPENSIM_SIMMKINEMATICSENGINE_MAJOR_VERSION}
-DOPENSIM_SIMMKINEMATICSENGINE_MINOR_VERSION=${OPENSIM_SIMMKINEMATICSENGINE_MINOR_VERSION}
-DOPENSIM_SIMMKINEMATICSENGINE_BUILD_VERSION=${OPENSIM_SIMMKINEMATICSENGINE_BUILD_VERSION}
-DOPENSIM_SIMMKINEMATICSENGINE_COPYRIGHT_YEARS="${OPENSIM_SIMMKINEMATICSENGINE_COPYRIGHT_YEARS}"
-DOPENSIM_SIMMKINEMATICSENGINE_AUTHORS="${OPENSIM_SIMMKINEMATICSENGINE_AUTHORS}"
-DOPENSIM_SIMMKINEMATICSENGINE_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} ${OpenSim_SOURCE_DIR}/Vendors ${XERCES_INCLUDE_DIR})

SET(EXPORT_MACRO OSIM${UKIT}_EXPORTS)

LINK_LIBRARIES( osimCommon osimSimulation osimActuators )

IF (BUILD_STATIC_OSIM_LIBS)
	ADD_LIBRARY(osim${KIT} STATIC ${SOURCE_FILES} ${INCLUDE_FILES}) 
	ADD_DEFINITIONS(-DSTATIC_OSIM_LIBS)
ELSE(BUILD_STATIC_OSIM_LIBS)
	ADD_LIBRARY(osim${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 
	SET_TARGET_PROPERTIES(osim${KIT} PROPERTIES
		DEFINE_SYMBOL ${EXPORT_MACRO}
		PROJECT_LABEL "Libraries - osim${KIT}"
	)
ENDIF(BUILD_STATIC_OSIM_LIBS)

INSTALL_TARGETS(/sdk/lib/ RUNTIME_DIRECTORY /bin osim${KIT})
