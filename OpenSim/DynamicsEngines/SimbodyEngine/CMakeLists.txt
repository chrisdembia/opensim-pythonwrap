SET(KIT SimbodyEngine)
SET(UKIT SIMBODYENGINE)

##
## Version stuff
##
SET(OPENSIM_SIMBODYENGINE_LIBRARY_NAME osim${KIT})
SET(OPENSIM_SIMBODYENGINE_MAJOR_VERSION 0)
SET(OPENSIM_SIMBODYENGINE_MINOR_VERSION 1)
SET(OPENSIM_SIMBODYENGINE_BUILD_VERSION 1)
SET(OPENSIM_SIMBODYENGINE_COPYRIGHT_YEARS "2007")
# comma separated list of authors
SET(OPENSIM_SIMBODYENGINE_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_SIMBODYENGINE_LIBRARY_NAME=${OPENSIM_SIMBODYENGINE_LIBRARY_NAME}
-DOPENSIM_SIMBODYENGINE_MAJOR_VERSION=${OPENSIM_SIMBODYENGINE_MAJOR_VERSION}
-DOPENSIM_SIMBODYENGINE_MINOR_VERSION=${OPENSIM_SIMBODYENGINE_MINOR_VERSION}
-DOPENSIM_SIMBODYENGINE_BUILD_VERSION=${OPENSIM_SIMBODYENGINE_BUILD_VERSION}
-DOPENSIM_SIMBODYENGINE_COPYRIGHT_YEARS="${OPENSIM_SIMBODYENGINE_COPYRIGHT_YEARS}"
-DOPENSIM_SIMBODYENGINE_AUTHORS="${OPENSIM_SIMBODYENGINE_AUTHORS}"
-DOPENSIM_SIMBODYENGINE_TYPE="Shared")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h )

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR}
		    ${OpenSim_SOURCE_DIR}/Vendors
			 ${XERCES_INCLUDE_DIR})

LINK_LIBRARIES( osimCommon osimSimulation )
ADD_LIBRARY(osim${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

ADD_DEFINITIONS("-DOSIM${UKIT}_EXPORTS")

INSTALL_TARGETS(/lib/ osim${KIT})
INSTALL_FILES(/include/OpenSim/DynamicsEngines/SimbodyEngine .h ${INCLUDE_FILES})
