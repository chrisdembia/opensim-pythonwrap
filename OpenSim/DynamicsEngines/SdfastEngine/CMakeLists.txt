SET(KIT SdfastEngine)
SET(UKIT SDFASTENGINE)

##
## Version stuff
##
SET(OPENSIM_SDFASTENGINE_LIBRARY_NAME osim${KIT})
SET(OPENSIM_SDFASTENGINE_MAJOR_VERSION 0)
SET(OPENSIM_SDFASTENGINE_MINOR_VERSION 1)
SET(OPENSIM_SDFASTENGINE_BUILD_VERSION 1)
SET(OPENSIM_SDFASTENGINE_COPYRIGHT_YEARS "2006")
# comma separated list of authors
SET(OPENSIM_SDFASTENGINE_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_SDFASTENGINE_LIBRARY_NAME=${OPENSIM_SDFASTENGINE_LIBRARY_NAME}
-DOPENSIM_SDFASTENGINE_MAJOR_VERSION=${OPENSIM_SDFASTENGINE_MAJOR_VERSION}
-DOPENSIM_SDFASTENGINE_MINOR_VERSION=${OPENSIM_SDFASTENGINE_MINOR_VERSION}
-DOPENSIM_SDFASTENGINE_BUILD_VERSION=${OPENSIM_SDFASTENGINE_BUILD_VERSION}
-DOPENSIM_SDFASTENGINE_COPYRIGHT_YEARS="${OPENSIM_SDFASTENGINE_COPYRIGHT_YEARS}"
-DOPENSIM_SDFASTENGINE_AUTHORS="${OPENSIM_SDFASTENGINE_AUTHORS}"
-DOPENSIM_SDFASTENGINE_TYPE="Static")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h )

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR}
		    ${OpenSim_SOURCE_DIR}/Vendors
			 ${XERCES_INCLUDE_DIR})

ADD_LIBRARY(osim${KIT} ${SOURCE_FILES} ${INCLUDE_FILES}) 

# For linux, gait23 doesn't link properly with SdfastEngine unless -fPIC is used
IF(NOT WIN32)
SET_TARGET_PROPERTIES(osim${KIT} PROPERTIES COMPILE_FLAGS -fPIC)
ENDIF(NOT WIN32)

ADD_DEFINITIONS("-DOSIM${UKIT}_EXPORTS")

INSTALL_TARGETS(/lib/ osim${KIT})
INSTALL_FILES(/include/OpenSim/DynamicsEngines/SdfastEngine .h ${INCLUDE_FILES})
