SET(KIT SdfastEngine)

##
## Version stuff
##
SET(OPENSIM_SDFAST_ENGINE_LIBRARY_NAME ${KIT})
SET(OPENSIM_SDFAST_ENGINE_MAJOR_VERSION 0)
SET(OPENSIM_SDFAST_ENGINE_MINOR_VERSION 1)
SET(OPENSIM_SDFAST_ENGINE_BUILD_VERSION 1)
SET(OPENSIM_SDFAST_ENGINE_COPYRIGHT_YEARS "2006")
# comma separated list of authors
SET(OPENSIM_SDFAST_ENGINE_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_SDFAST_ENGINE_LIBRARY_NAME=${OPENSIM_SDFAST_ENGINE_LIBRARY_NAME}
-DOPENSIM_SDFAST_ENGINE_MAJOR_VERSION=${OPENSIM_SDFAST_ENGINE_MAJOR_VERSION}
-DOPENSIM_SDFAST_ENGINE_MINOR_VERSION=${OPENSIM_SDFAST_ENGINE_MINOR_VERSION}
-DOPENSIM_SDFAST_ENGINE_BUILD_VERSION=${OPENSIM_SDFAST_ENGINE_BUILD_VERSION}
-DOPENSIM_SDFAST_ENGINE_COPYRIGHT_YEARS="${OPENSIM_SDFAST_ENGINE_COPYRIGHT_YEARS}"
-DOPENSIM_SDFAST_ENGINE_AUTHORS="${OPENSIM_SDFAST_ENGINE_AUTHORS}"
-DOPENSIM_SDFAST_ENGINE_TYPE="Static")

FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES *.h )

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} 
		    ${OpenSim_SOURCE_DIR}/Vendors 		    		    
		    ${OpenSim_SOURCE_DIR}/OpenSim/Models/SdfastEngine
			 ${XERCES_INCLUDE_DIR}
		    ${OpenSim_SOURCE_DIR}/Tools)

ADD_LIBRARY(${KIT} ${SOURCE_FILES} ${INCLUDE_FILES}) 

# For linux, gait23 doesn't link properly with SdfastEngine unless -fPIC is used
IF(NOT WIN32)
SET_TARGET_PROPERTIES(${KIT} PROPERTIES COMPILE_FLAGS -fPIC)
ENDIF(NOT WIN32)

ADD_DEFINITIONS("-DSDFAST_ENGINE_EXPORTS -D_USRDLL")

INSTALL_TARGETS(/lib/ ${KIT})
INSTALL_FILES(/include/OpenSim/Models/SdfastEngine .h ${INCLUDE_FILES})
