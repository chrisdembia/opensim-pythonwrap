SET(KIT Tools)
SET(UKIT TOOLS)

##
## Version stuff
##
SET(OPENSIM_RDTOOLS_LIBRARY_NAME rd${KIT})
SET(OPENSIM_RDTOOLS_MAJOR_VERSION 0)
SET(OPENSIM_RDTOOLS_MINOR_VERSION 6)
SET(OPENSIM_RDTOOLS_BUILD_VERSION 1)
SET(OPENSIM_RDTOOLS_COPYRIGHT_YEARS "2005-6")
# comma separated list of authors
SET(OPENSIM_RDTOOLS_AUTHORS  "Clay_Anderson-Ayman_Habib_and_Peter_Loan") 


ADD_DEFINITIONS(-DOPENSIM_RDTOOLS_LIBRARY_NAME=${OPENSIM_RDTOOLS_LIBRARY_NAME}
-DOPENSIM_RDTOOLS_MAJOR_VERSION=${OPENSIM_RDTOOLS_MAJOR_VERSION}
-DOPENSIM_RDTOOLS_MINOR_VERSION=${OPENSIM_RDTOOLS_MINOR_VERSION}
-DOPENSIM_RDTOOLS_BUILD_VERSION=${OPENSIM_RDTOOLS_BUILD_VERSION}
-DOPENSIM_RDTOOLS_COPYRIGHT_YEARS="${OPENSIM_RDTOOLS_COPYRIGHT_YEARS}"
-DOPENSIM_RDTOOLS_AUTHORS="${OPENSIM_RDTOOLS_AUTHORS}"
-DOPENSIM_RDTOOLS_TYPE="Shared")


FILE(GLOB SOURCE_FILES *.cpp gcvspl.c)
FILE(GLOB INCLUDE_FILES *.h gcvspl.h)

INCLUDE_DIRECTORIES(${OpenSim_SOURCE_DIR} ${OpenSim_SOURCE_DIR}/Vendors ${OpenSim_SOURCE_DIR}/Vendors/xerces-c-src2_4_0/src)

SET(EXPORT_MACRO RD${UKIT}_EXPORTS)

LINK_LIBRARIES( debug xerces-c_2D optimized  xerces-c_2)

ADD_LIBRARY(rd${KIT} SHARED ${SOURCE_FILES} ${INCLUDE_FILES}) 

IF(WIN32)
	ADD_DEPENDENCIES(rd${KIT} xerces-c_2D)
ENDIF(WIN32)

ADD_DEPENDENCIES(rd${KIT} xerces-c_2)

SET_TARGET_PROPERTIES(rd${KIT} PROPERTIES
   DEFINE_SYMBOL ${EXPORT_MACRO}
)

INSTALL_TARGETS(/lib/ RUNTIME_DIRECTORY /lib rd${KIT})

INSTALL_FILES(/include/OpenSim/Tools .h ${INCLUDE_FILES})
